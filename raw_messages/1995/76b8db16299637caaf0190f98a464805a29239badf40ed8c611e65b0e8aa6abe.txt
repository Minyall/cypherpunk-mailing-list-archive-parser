(defun modexpt (x y n)  "computes (x^y) mod n"
  (cond ((= y 0) 1)   ((= y 1) (mod x n))
	((evenp y) (mod (expt (modexpt x (/ y 2) n) 2) n))
	(t (mod (* x (modexpt x (1- y) n)) n))))
